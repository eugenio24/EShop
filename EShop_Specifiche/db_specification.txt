user
 - id PK
 - name
 - surname
 - username
 - email
 - psw_hash
 - psw_salt
 - registration_date
 - registration_type enum('NATIVE', 'GOOGLE', ecc.)
 - external_id (id di google)

product
 - id PK
 - name
 - desc
 - price
 - category FK->product_category
 - image_path (le immagini dei prodotti vengono salvate nel file-system (o una cdn?) )

product_category
 - id PK
 - name

order_head
 - order_number PK
 - idUser FK->user
 - order_date
 - total_price 
 - shipping_address
 
order_row
 - id PK
 - idOrder FK->order_head
 - idProduct FK->product
 - quantity 
 - price (singolo, non per tutta la quantità)
